RcppRegression Benchmark

Small Matrix [4c x 11r]


vec <- as.numeric(cereal[, 2])
mat <- as.matrix(cereal[, 3:6])
lm <- new(LS, vec, mat)


res <- microbenchmark(lm$BestSubset(), lm$ompBestSubset(), times = 1000L)

print(res)

## Unit: microseconds
##                expr min    lq median    uq  max neval
##     lm$BestSubset() 141 207.8  211.9 215.0 2146  1000
##  lm$ompBestSubset() 138 149.1  154.4 160.1 5271  1000


autoplot(res)
